```typescript
it('no service_role or superuser references in API routes', () => {
  // Walk apps/web/app/api/ recursively and collect all .ts/.tsx file contents
  const apiDir = join(__dirname, '../../../../apps/web/app/api')

  function getAllFiles(dir: string): string[] {
    const entries = readdirSync(dir, { withFileTypes: true })
    return entries.flatMap(entry => {
      const fullPath = join(dir, entry.name)
      return entry.isDirectory() ? getAllFiles(fullPath) : [fullPath]
    })
  }

  const files = getAllFiles(apiDir).filter(f => f.endsWith('.ts') || f.endsWith('.tsx'))
  const forbidden = ['service_role', 'SUPABASE_SERVICE_ROLE', 'postgresql://postgres:postgres@']

  for (const file of files) {
    const content = readFileSync(file, 'utf-8')
    for (const term of forbidden) {
      expect(
        content,
        `Found forbidden term "${term}" in ${file}`
      ).not.toContain(term)
    }
  }
})
```

Verify this relative path is correct for the monorepo layout before writing.

The forbidden terms list mirrors the CI grep pattern: `service_role` covers both `SUPABASE_SERVICE_ROLE_KEY` and any usage of that key name; `postgresql://postgres:postgres@` catches hardcoded superuser connection strings.

Do NOT make the test async — readdirSync and readFileSync are synchronous. The `async` keyword on the original stub can be removed.
