    ```typescript
    export interface DispatchLink {
      id: string
      dispatch_date: Date
      destination: string
      status: string
    }

    export interface TraceabilityChain {
      pickup: { id: string; reference: string; status: string; created_at: Date } | null
      transport: { id: string; type: string; provider?: string; status: string } | null
      intake: { id: string; reference: string; staff_name: string; delivery_date: Date; is_unexpected: boolean }
      wash: { id: string; staff_name: string; report_date: Date; product_name: string } | null
      pack: { id: string; staff_name: string; report_date: Date; product_name: string } | null
      dispatch: DispatchLink | null          // Deterministic (via intake_record_id FK)
      dispatchFallback: DispatchLink[] | null // Non-deterministic fallback (via facility); null when dispatch is set
    }

    // Pure type + helper for building chain from raw query results
    export function assembleTraceabilityChain(data: {
      intake: { ... }
      pickup: { ... } | null
      transport: { ... } | null
      washReports: Array<{ ... }>
      packReports: Array<{ ... }>
      directDispatch: { ... } | null          // Found via intake_record_id FK
      facilityDispatches: Array<{ ... }>      // Found via facility fallback (only used when directDispatch is null)
    }): TraceabilityChain {
      return {
        pickup: data.pickup,
        transport: data.transport,
        intake: data.intake,
        wash: data.washReports[0] ?? null,
        pack: data.packReports[0] ?? null,
        dispatch: data.directDispatch,
        dispatchFallback: data.directDispatch ? null : (data.facilityDispatches.length > 0 ? data.facilityDispatches : null),
      }
    }
    ```
