    ```typescript
    export interface TraceabilityChain {
      pickup: { id: string; reference: string; status: string; created_at: Date } | null
      transport: { id: string; type: string; provider?: string; status: string } | null
      intake: { id: string; reference: string; staff_name: string; delivery_date: Date; is_unexpected: boolean }
      wash: { id: string; staff_name: string; report_date: Date; product_name: string } | null
      pack: { id: string; staff_name: string; report_date: Date; product_name: string } | null
      dispatch: { id: string; dispatch_date: Date; destination: string; status: string } | null
    }

    // Pure type + helper for building chain from raw query results
    export function assembleTraceabilityChain(data: {
      intake: { ... }
      pickup: { ... } | null
      transport: { ... } | null
      washReports: Array<{ ... }>
      packReports: Array<{ ... }>
      dispatch: { ... } | null
    }): TraceabilityChain {
      return {
        pickup: data.pickup,
        transport: data.transport,
        intake: data.intake,
        wash: data.washReports[0] ?? null,   // Most recent wash report
        pack: data.packReports[0] ?? null,   // Most recent pack report
        dispatch: data.dispatch,
      }
    }
    ```
