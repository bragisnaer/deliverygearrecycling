    ```typescript
    export async function voidIntakeRecord(id: string, reason: string) {
      const user = await requireRecoAdmin()

      const validation = validateVoidInput(reason)
      if (!validation.valid) return { error: validation.error }

      const [record] = await withRLSContext(user, (tx) =>
        tx.select({ voided: intakeRecords.voided })
          .from(intakeRecords)
          .where(eq(intakeRecords.id, id))
          .limit(1)
      )

      if (!record) return { error: 'not_found' }
      if (record.voided) return { error: 'already_voided' }

      await withRLSContext(user, (tx) =>
        tx.update(intakeRecords)
          .set({ voided: true, void_reason: reason, updated_at: new Date() })
          .where(eq(intakeRecords.id, id))
      )

      return { success: true }
    }
    ```

    Also update any existing list/query functions in intake actions to add `.where(eq(intakeRecords.voided, false))` filter — check existing getIntakeQueue or similar functions and add the voided filter.
